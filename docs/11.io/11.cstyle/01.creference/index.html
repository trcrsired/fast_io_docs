Ch11.11.1: Complete C I/O Function Reference

File Opening and Closing

Function Description
fopenOpen a file
fcloseClose a file
fflushFlush a stream’s buffer
 

#include <stdio.h>

int main() {
    FILE *fp = fopen("test.txt", "w");
    if (fp == NULL) {
        perror("fopen failed");
        return 1;
    }
    fprintf(fp, "Hello, World!\n");
    fflush(fp);  // Force write to disk
    fclose(fp);
    return 0;
}

Formatted Output

Function Description
printfFormatted output to stdout
fprintfFormatted output to a stream
sprintfFormatted output to a string buffer
snprintfFormatted output to a string buffer with size limit
vprintfFormatted output to stdout (variadic)
vfprintfFormatted output to a stream (variadic)
vsprintfFormatted output to a string buffer (variadic)
vsnprintfFormatted output to a string buffer with size limit (variadic)
 

#include <stdio.h>

int main() {
    int x = 42;
    double pi = 3.14159;

    // printf to stdout
    printf("x = %d, pi = %.2f\n", x, pi);

    // fprintf to file
    FILE *fp = fopen("output.txt", "w");
    fprintf(fp, "x = %d\n", x);
    fclose(fp);

    // sprintf to string buffer
    char buffer[100];
    sprintf(buffer, "Value: %d", x);

    // snprintf with size limit (safer)
    snprintf(buffer, sizeof(buffer), "Value: %d", x);

    return 0;
}

Formatted Input

Function Description
scanfFormatted input from stdin
fscanfFormatted input from a stream
sscanfFormatted input from a string
 

#include <stdio.h>

int main() {
    int x;
    double y;

    // scanf from stdin
    printf("Enter two numbers: ");
    scanf("%d %lf", &x, &y);

    // fscanf from file
    FILE *fp = fopen("data.txt", "r");
    if (fp) {
        fscanf(fp, "%d %lf", &x, &y);
        fclose(fp);
    }

    // sscanf from string
    char *str = "42 3.14";
    sscanf(str, "%d %lf", &x, &y);

    return 0;
}

Character and Line I/O

Function Description
fgetc / getcRead a character from a stream
fgetsRead a line from a stream
fputc / putcWrite a character to a stream
fputsWrite a string to a stream
putsWrite a string to stdout (adds newline)
ungetcPush a character back onto a stream
getsRead a line from stdin (deprecated, unsafe)
 

#include <stdio.h>

int main() {
    // Write characters
    FILE *fp = fopen("test.txt", "w");
    fputc('A', fp);
    fputs("Hello\n", fp);
    fclose(fp);

    // Read characters
    fp = fopen("test.txt", "r");
    int ch = fgetc(fp);
    printf("First char: %c\n", ch);

    // Read line
    char buffer[100];
    fgets(buffer, sizeof(buffer), fp);
    printf("Line: %s", buffer);

    // Push back
    ungetc(ch, fp);

    fclose(fp);

    // puts to stdout
    puts("Hello, World!");  // Adds newline automatically

    return 0;
}

Binary I/O

Function Description
freadRead binary data from a stream
fwriteWrite binary data to a stream
 

#include <stdio.h>

struct Point {
    int x;
    int y;
};

int main() {
    struct Point p1 = {10, 20};
    struct Point p2;

    // Write binary data
    FILE *fp = fopen("points.bin", "wb");
    fwrite(&p1, sizeof(struct Point), 1, fp);
    fclose(fp);

    // Read binary data
    fp = fopen("points.bin", "rb");
    fread(&p2, sizeof(struct Point), 1, fp);
    fclose(fp);

    printf("Read point: (%d, %d)\n", p2.x, p2.y);

    return 0;
}

File Positioning

Function Description
fseekSet file position
ftellGet current file position
rewindSet file position to beginning
fgetposGet file position (portable)
fsetposSet file position (portable)
 

#include <stdio.h>

int main() {
    FILE *fp = fopen("test.txt", "w+");
    fputs("Hello, World!", fp);

    // Get current position
    long pos = ftell(fp);
    printf("Current position: %ld\n", pos);

    // Seek to beginning
    rewind(fp);

    // Or use fseek
    fseek(fp, 0, SEEK_SET);  // Beginning
    fseek(fp, 0, SEEK_END);  // End
    fseek(fp, 5, SEEK_SET);  // Position 5

    // Portable positioning
    fpos_t position;
    fgetpos(fp, &position);
    // ... do something ...
    fsetpos(fp, &position);  // Restore position

    fclose(fp);
    return 0;
}

Error Handling

Function Description
feofTest for end-of-file
ferrorTest for error
clearerrClear error and end-of-file indicators
perrorPrint error message to stderr
 

#include <stdio.h>

int main() {
    FILE *fp = fopen("test.txt", "r");
    if (fp == NULL) {
        perror("Failed to open file");  // Prints error message
        return 1;
    }

    int ch;
    while ((ch = fgetc(fp)) != EOF) {
        putchar(ch);
    }

    if (feof(fp)) {
        printf("\nEnd of file reached\n");
    }
    if (ferror(fp)) {
        printf("\nError occurred\n");
    }

    clearerr(fp);  // Clear error/EOF flags
    fclose(fp);
    return 0;
}

File Management

Function Description
removeDelete a file
renameRename a file
tmpfileCreate a temporary file
tmpnamGenerate a temporary filename (unsafe)
 

#include <stdio.h>

int main() {
    // Rename a file
    rename("old.txt", "new.txt");

    // Delete a file
    remove("unwanted.txt");

    // Create temporary file (automatically deleted on close)
    FILE *tmp = tmpfile();
    if (tmp) {
        fprintf(tmp, "Temporary data\n");
        fclose(tmp);  // File is automatically deleted
    }

    return 0;
}

Buffering Control

Function Description
setvbufSet stream buffering mode
setbufSet stream buffer
 

#include <stdio.h>

int main() {
    FILE *fp = fopen("test.txt", "w");

    char buffer[1024];

    // Set custom buffer
    setvbuf(fp, buffer, _IOFBF, sizeof(buffer));  // Full buffering
    setvbuf(fp, NULL, _IOLBF, 0);  // Line buffering
    setvbuf(fp, NULL, _IONBF, 0);  // No buffering

    // Simpler interface
    setbuf(fp, buffer);  // Equivalent to setvbuf(fp, buffer, _IOFBF, BUFSIZ)
    setbuf(fp, NULL);    // Unbuffered

    fclose(fp);
    return 0;
}
